 Write a Python program to prepare Scatter Plot for Iris Dataset. Convert Categorical values in numeric format for a dataset. 
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.preprocessing import LabelEncoder

# Step 1: Load the Iris datasetpip 
iris = load_iris()
iris_data = pd.DataFrame(iris.data, columns=iris.feature_names)
iris_data['species'] = iris.target

# Step 2: Convert categorical 'species' column to numeric using LabelEncoder
label_encoder = LabelEncoder()
iris_data['species'] = label_encoder.fit_transform(iris_data['species'])

# Step 3: Scatter Plot (using sepal length and sepal width as an example)
plt.figure(figsize=(8, 6))
sns.scatterplot(data=iris_data, x='sepal length (cm)', y='sepal width (cm)', hue='species', palette='Set1', style='species', s=100)

# Customizing the plot
plt.title("Scatter Plot of Iris Dataset")
plt.xlabel('Sepal Length (cm)')
plt.ylabel('Sepal Width (cm)')

# Display the plot
plt.show()
